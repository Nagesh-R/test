ALTER PROC [hotels].[sp_Getroomtype]  
@HotelID numeric(18,0),
@isFlag varchar(1)
AS    
BEGIN 
if(@isFlag = 'Y' or @isFlag = 'y')
begin   
      select Id_room,Code_room,Desc_room from b_room where id_hotel=@HotelID and UPPER(flg_sale) = 'Y'
      and No_room > 0 
end 
else
begin
      select Id_room,Code_room,Desc_room from b_room where id_hotel=@HotelID  
end 
END
